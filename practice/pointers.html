    Pointers

  1. 1. What does following code output?

    
void coutA(int*a){
    cout << a << " | " << *a << " | in pointer func" << endl;
}
void coutA(int&a){
    cout << &a << " | " << a << " | in reference func" << endl;
}
                    
int main() {
    int a = 42;
    int *pA = &a;
    int **pA2 = &pA;

    coutA(a);
    coutA(&a);
    coutA(pA);
    coutA(*pA);
    coutA(*pA2);
    coutA(**pA2);

    cout << endl;
    cout << &a << endl;
    cout << &pA << endl;
    cout << &pA2 << endl;
